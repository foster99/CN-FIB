function y = f(x, n)
% Evalua la funcion f en x con una aritmetica de n digitos por redondeo
R = @(x) myRound(x,n);
% Calculo de las potencias de euler
% [Ejemplo: e^(4x) -> (e^x)^4 -> (e^x)(e^x)(e^x)(e^x)]
e1 = R(R(exp(1)) ^ x);
e2 = R(e1 * e1);
e3 = R(e2 * e1);
e4 = R(e3 * e1);
% Calculo de los productos
a = R(1.01 * e4);
b = R(4.62 * e3);
c = R(3.11 * e2);
d = R(12.2 * e1);
% Calculo de sumas y restas
y = R(R(R(R(a - b) - c) + d) - 1.99);
end
function y = F(x, n)
% Evalua la funcion F en x con una aritmetica de n digitos por redondeo
R = @(x) myRound(x,n);
z = R(R(exp(1)) ^ x);
a = R(1.01 * z);
b = R(R(a - 4.62) * z);
c = R(R(b - 3.11) * z);
d = R(R(c + 12.2) * z);
y = R(d - 1.99);
end
