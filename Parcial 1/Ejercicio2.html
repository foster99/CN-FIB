function y = p_horner(x, n)
% Evalua el polinomio P en x con una aritmetica de n digitos por redondeo
R = @(x) myRound(x,n);
X = R(x);
e1 = R(X - R(6.1));
e2 = R(X * e1);
e3 = R(e2 + R(3.2));
e4 = R(X * e3);
y = R(e4 + R(1.5));
end
function y = p(x, n)
% Evalua el polinomio P en x con una aritmetica de n digitos por redondeo
R = @(x) myRound(x,n);
X = R(x);
square = R(X*X);
cube = R(square*X);
e3 = cube;
e2 = R(R(6.1)* square);
e1 = R(R(3.2) * X);
e0 = R(1.5);
% Calculo de sumas y restas
y = R(R(R(e3 - e2) + e1) + e0);
end
function y = myRound(x,t)
y = round(x,t,'significant');
end
